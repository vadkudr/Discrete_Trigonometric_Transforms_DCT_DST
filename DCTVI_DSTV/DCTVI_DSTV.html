Definitions

alpha=sort([roots(TschebyshevW(N-1)); 1],'descend');
DCT6t=zeros(N1);
for l=0:N1-1,
    DCT6t(:,l+1)=polyval(TschebyshevV(l),alpha)';
end
Da6=diag(cos(1/2*pi/(N-1/2)*k));
DCT6t=Da6*DCT6t                    % display DCTV matrix
